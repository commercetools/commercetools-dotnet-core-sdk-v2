95 public static IHttpClientBuilder SetupClient(this IServiceCollection services, string clientName, Func<HttpResponseMessage, Type> errorResponseTypeMapper, Func<IServiceProvider, ISerializerService> serializerFactory, ClientOptions options = null)
96 {
97 options ??= new ClientOptions();
98 services.AddSingleton<IUserAgentProvider, UserAgentProvider>();
99 var httpClientBuilder = services.AddHttpClient(clientName)

101 {
102 if (options.DecompressionMethods.HasFlag(DecompressionMethods.GZip))
103 {
104 client.DefaultRequestHeaders.AcceptEncoding.ParseAdd("gzip");
105 }
106 if (options.DecompressionMethods.HasFlag(DecompressionMethods.Deflate))
107 {
108 client.DefaultRequestHeaders.AcceptEncoding.ParseAdd("deflate");
109 }
110 var userAgentProvider = provider.GetService<IUserAgentProvider>() ?? new UserAgentProvider();
111 client.DefaultRequestHeaders.UserAgent.ParseAdd(userAgentProvider.UserAgent);
112 })
113 .ConfigureHttpMessageHandlerBuilder(builder =>

118 };
